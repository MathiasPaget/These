	\foreach \i in {0,...,4}{
		\pgfmathtruncatemacro{\x}{\i-2}
		\pgfmathtruncatemacro{\a}{\i-1}
		\foreach \j in {0,...,4}{
			\pgfmathtruncatemacro{\y}{\j-2}
			\pgfmathtruncatemacro{\b}{\j-1}
			
			\node[main node] at (\i,\j) (\i\j) {};
			
			\ifthenelse{\i>0}{
				\path[-] (\a\j) edge node[left] {} (\i\j);
			}
			
			\ifthenelse{\j>0}{
				\path[-] (\i\b) edge node[left] {} (\i\j);
			}			
		}
	}
\end{tikzpicture}
